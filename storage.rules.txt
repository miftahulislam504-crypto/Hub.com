// Firebase Storage Rules
// Firebase Console → Storage → Rules এ paste করুন

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    match /projects/{projectId}/documents/{fileName} {
      // Authenticated users can read/write their project documents
      allow read:  if request.auth != null;
      allow write: if request.auth != null
                   && request.resource.size < 20 * 1024 * 1024  // 20 MB max
                   && (
                     request.resource.contentType.matches('image/.*') ||
                     request.resource.contentType.matches('application/pdf') ||
                     request.resource.contentType.matches('application/.*') ||
                     request.resource.contentType.matches('text/.*')
                   );
      allow delete: if request.auth != null;
    }
  }
}
