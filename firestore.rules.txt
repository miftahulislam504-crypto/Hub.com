// firestore.rules.txt
// Firebase Console → Firestore → Rules → এটা paste করুন → Publish

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Users — নিজের data নিজে পড়বে/লিখবে
    match /users/{userId} {
      allow read, write: if request.auth != null
                         && request.auth.uid == userId;
    }

    // Projects — নিজের project নিজে manage করবে
    match /projects/{projectId} {
      allow read, write: if request.auth != null
                         && request.auth.uid == resource.data.createdBy;

      allow create: if request.auth != null
                    && request.auth.uid == request.resource.data.createdBy;

      // Sub-collections — project owner পড়বে/লিখবে
      match /{subCollection}/{docId} {
        allow read, write: if request.auth != null
                           && get(/databases/$(database)/documents/projects/$(projectId)).data.createdBy
                              == request.auth.uid;
      }
    }
  }
}
